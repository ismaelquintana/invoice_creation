================================================================================
3. FILE FORMAT ANALYSIS
================================================================================

RECOMMENDED: JSON
  Pros:
    - Human-readable (debug-friendly)
    - Industry standard
    - Elixir has excellent support (Jason library)
    - Easy versioning and migration
    - Works with APIs, databases, tools
    - Standard for Phoenix/web frameworks

  Cons:
    - No built-in type safety
    - Requires parsing

ALTERNATIVES CONSIDERED:
  Erlang Binary (:erlang.term_to_binary)
    - Fast and space-efficient
    - Preserves types exactly
    - Cons: Not human-readable, Erlang-specific

  TOML
    - Readable minimal syntax
    - Cons: Less common, slower

  Protocol Buffers
    - Compact and versioned
    - Cons: Complex tooling, binary format

  ✓ Year-based organization aligns with ListInvoiceYear structure
  ✓ Simplifies archive/purge operations
  ✓ Good filesystem performance

5. PUBLIC API DESIGN
================================================================================

InvoiceStorage Module (Main API):

  Single Operations:
    save(invoice :: Invoice.t()) -> :ok | {:error, Error.t()}
    load(number :: String.t()) -> {:ok, Invoice.t()} | {:error, Error.t()}
    delete(number :: String.t()) -> :ok | {:error, Error.t()}
    exists?(number :: String.t()) -> boolean()

  Batch Operations:
    save_all(invoices :: [Invoice.t()]) -> :ok | {:error, Error.t()}
    load_all(year :: pos_integer()) -> {:ok, [Invoice.t()]} | {:error, Error.t()}

  Year List Operations:
    save_year_list(list :: ListInvoiceYear.t()) -> :ok | {:error, Error.t()}
    load_year_list(year :: pos_integer()) -> {:ok, ListInvoiceYear.t()} | {:error, Error.t()}

  Utility Operations:
    list_years() -> {:ok, [pos_integer()]} | {:error, Error.t()}
    count(year :: pos_integer()) -> {:ok, non_neg_integer()} | {:error, Error.t()}

InvoiceStorage.Encoder Module:
  encode_invoice(Invoice.t()) -> {:ok, String.t()} | {:error, Error.t()}
  encode_list_invoice_year(ListInvoiceYear.t()) -> {:ok, String.t()} | {:error, Error.t()}
  encode_item(Item.t()) -> {:ok, String.t()} | {:error, Error.t()}

InvoiceStorage.Decoder Module:
  decode_invoice(String.t()) -> {:ok, Invoice.t()} | {:error, Error.t()}
  decode_list_invoice_year(String.t()) -> {:ok, ListInvoiceYear.t()} | {:error, Error.t()}
  decode_item(String.t()) -> {:ok, Item.t()} | {:error, Error.t()}

6. ERROR HANDLING STRATEGY
================================================================================

InvoiceStorage.Error Type Definition:
  %{type: error_type(), message: String.t(), context: map()}

Error Types:
  :file_not_found         - Invoice file doesn't exist
  :permission_denied      - Can't access file/directory
  :disk_full              - No space to write
  :invalid_path           - Invalid path construction
  :encode_failed          - Encoding to JSON failed
  :decode_failed          - Decoding from JSON failed
  :invalid_json           - JSON syntax error
  :invalid_invoice_data   - Validation failed after deserialization
  :invalid_year           - Year validation failed
  :io_error               - Generic I/O error
  :unknown_error          - Unexpected error

ERROR HANDLING PATTERNS:

  1. Pattern Matching:
     case InvoiceStorage.load(number) do
       {:ok, invoice} -> process(invoice)
       {:error, error} -> handle_error(error)
     end

  2. Error Recovery:
     case error.type do
       :file_not_found -> "Not found"
       :permission_denied -> "Access denied"
       :decode_failed -> "Corrupted data"
       _ -> "Unknown error"
     end

  3. Retry Logic:
     def save_with_retry(invoice, retries \\ 3) do
       case save(invoice) do
         :ok -> :ok
         {:error, error} when retries > 0 ->
           Process.sleep(100)
           save_with_retry(invoice, retries - 1)
         {:error, error} -> {:error, error}
       end
     end

  {:ok, invoice} = Invoice.add_item(invoice, item)
  :ok = InvoiceStorage.save(invoice)

Integration with Invoice.update/1:
  {:ok, invoice} = InvoiceStorage.load(number)
  {:ok, updated} = Invoice.update(invoice, vat: 300)
  :ok = InvoiceStorage.save(updated)

Integration with ListInvoiceYear:
  {:ok, list} = ListInvoiceYear.new(year: 2024)
  {:ok, list} = ListInvoiceYear.add_invoice(list, invoice)
  :ok = InvoiceStorage.save_year_list(list)
  :ok = InvoiceStorage.save(invoice)

Repository Pattern (Recommended):
  defmodule InvoiceRepository do
    def create(opts) do
      with {:ok, invoice} <- Invoice.new(opts),
           :ok <- InvoiceStorage.save(invoice) do
        {:ok, invoice}
      end
    end

    def get(number), do: InvoiceStorage.load(number)

    def update(number, updates) do
      with {:ok, invoice} <- InvoiceStorage.load(number),
           {:ok, updated} <- Invoice.update(invoice, updates),
           :ok <- InvoiceStorage.save(updated) do
        {:ok, updated}
      end
    end

    def delete(number) do
      InvoiceStorage.delete(number)
    end

    def list_by_year(year) do
      InvoiceStorage.load_all(year)
    end
  end

14. COMMON PATTERNS
================================================================================

Pattern 1: Simple Save
  {:ok, invoice} = Invoice.new(opts)
  {:ok, invoice} = Invoice.add_item(invoice, item)
  :ok = InvoiceStorage.save(invoice)

Pattern 2: Load and Update
  {:ok, invoice} = InvoiceStorage.load(number)
  {:ok, updated} = Invoice.update(invoice, opts)
  :ok = InvoiceStorage.save(updated)

Pattern 3: Repository Pattern
  {:ok, invoice} = InvoiceRepository.create(opts)
  {:ok, invoice} = InvoiceRepository.get(number)
  :ok = InvoiceRepository.delete(number)

Pattern 4: Batch Processing
  {:ok, invoices} = InvoiceStorage.load_all(2024)
  processed = Enum.map(invoices, &process/1)
  :ok = InvoiceStorage.save_all(processed)

Pattern 5: Error Recovery
  case InvoiceStorage.load(number) do
    {:ok, invoice} -> {:ok, invoice}
    {:error, %{type: :file_not_found}} -> {:error, "Not found"}
    {:error, error} -> {:error, error.message}
  end

Pattern 6: Safe Load with Default
  invoice = InvoiceStorage.load(number)
            |> case do
              {:ok, inv} -> inv
              {:error, _} -> nil
            end
